  <!-- Script jQuery untuk menampilkan produk -->
    <script>
      // Fungsi untuk mendapatkan dan menampilkan produk berdasarkan jenis
      function loadProduk(jenisProduk) {
        // Lakukan request ke endpoint Flask untuk mendapatkan produk berdasarkan jenis
        $.ajax({
          url: `/produk`,
          type: 'GET',
          data: { jenis: jenisProduk }, // Parameter yang dikirimkan ke backend
          success: function (response) {
            let produkContainer = $('#produk-container');
            produkContainer.empty(); // Kosongkan container produk

            // Looping untuk setiap produk dan menampilkannya di halaman
            response.produk.forEach(function (produk) {
              let produkCard = `
                      <div class="col-md-4">
                          <div class="card mb-3 mt-5 m-3">
                              <div class="row g-0">
                                  <div class="col-md-4">
                                      <img src="/${produk.file}" alt="${produk.nama}" class="img-fluid rounded mt-2" />
                                  </div>
                                  <div class="col-md-8">
                                      <div class="card-body">
                                          <h5 class="card-title">${produk.nama}</h5>
                                          <p class="card-text">Stok: ${produk.stok}</p>
                                          <p class="card-text">Harga: Rp${produk.harga}</p>
                                      </div>
                                  </div>
                              </div>
                          </div>
                      </div>
                  `;
              produkContainer.append(produkCard); // Tambahkan produk ke kontainer
            });
          },
          error: function (error) {
            console.log('Error:', error);
          },
        });
      }

      // Event listener untuk dropdown
      $('#jenis-produk').change(function () {
        let jenisProduk = $(this).val(); // Ambil nilai dari dropdown
        loadProduk(jenisProduk); // Panggil fungsi untuk memuat produk
      });
    </script>
